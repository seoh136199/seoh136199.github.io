int cmp(const void *a, const void *b);
void inputArr(int *arr, int len);
void makePrtlSum(int *orgnArr, int *newArr, int len, int *pIdx);

void inputArr(int *arr, int len) {
	for (int i = 1; i <= len; i++) {
		scanf("%d", &arr[i]);
		arr[i] += arr[i - 1];
	}
	return;
}

void makePrtlSum(int *orgnArr, int *newArr, int len, int *pIdx) {
	for (int i = 1; i <= len; i++) {
		for (int j = 0; j < i; j++) {
			newArr[(*pIdx)++] = orgnArr[i] - orgnArr[j];
		}
	}
	return;
}

int cmp(const void *a, const void *b) {
	if (*(int *)a > *(int *)b) return 1;
	if (*(int *)a < *(int *)b) return -1;
	return 0;
}
